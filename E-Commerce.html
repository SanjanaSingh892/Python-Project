import seaborn as sns
import matplotlib.pyplot as plt

# Example Top 5 categories
Top5 = (
    sales.groupby("Product_Category")["Quantity"]
    .sum()
    .sort_values(ascending=False)
    .head(5)
    .reset_index()
)

# Create barplot
plt.figure(figsize=(8,5))
ax = sns.barplot(x="Product_Category", y="Quantity", data=Top5, palette="Blues_d")
plt.title("Top 5 Product Categories by Quantity")
plt.xticks(rotation=45)

# Add labels on top of bars
for p in ax.patches:
    ax.annotate(
        f'{p.get_height():.0f}',                      # text = bar height
        (p.get_x() + p.get_width()/2., p.get_height()), # position = center top of bar
        ha='center', va='bottom', fontsize=10, color='black'
    )

plt.show()
